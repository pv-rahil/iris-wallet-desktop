39class CFAViewModel(QObject, ThreadManager): 

40 """This class represents the activities of the bitcoin page.""" 

41 

42 asset_info = Signal(str, str, str, str) 

43 txn_list_loaded = Signal(str, str, str, str) 

44 send_cfa_button_clicked = Signal(bool) 

45 message = Signal(ToastPreset, str) 

46 is_loading = Signal(bool) 

47 refresh = Signal(bool) 

48 stop_loading = Signal(bool) 

49 

50 def __init__(self, page_navigation: Any) -> None: 

51 super().__init__() 

52 self._page_navigation = page_navigation 

53 self.asset_info.connect(self.get_cfa_asset_detail) 

54 

55 # Initializing default values for attributes 

56 self.asset_id = None 

57 self.asset_name = None 

58 self.image_path = None 

59 self.asset_type = None 

60 self.blinded_utxo = None 

61 self.transport_endpoints = None 

62 self.amount = None 

63 self.fee_rate = None 

64 self.min_confirmation = None 

65 self.txn_list = [] 

66 

67 def get_cfa_asset_detail(self, asset_id: str, asset_name: str, image_path: str, asset_type: str) -> None: 

68 """Retrieve CFA asset list.""" 

69 

70 def on_success(response: ListOnAndOffChainTransfersWithBalance) -> None: 

71 """Handle success for the CFA asset detail list.""" 

72 self.txn_list = response 

73 self.txn_list_loaded.emit( 

74 asset_id, asset_name, image_path, asset_type, 

75 ) 

76 self.asset_id, self.asset_name, self.image_path, self.asset_type = asset_id, asset_name, image_path, asset_type 

77 self.is_loading.emit(False) 

78 

79 def on_error(error: CommonException) -> None: 

80 """Handle error for the main asset page.""" 

81 self.txn_list_loaded.emit( 

82 asset_id, asset_name, image_path, asset_type, 

83 ) 

84 self.is_loading.emit(False) 

85 ToastManager.error(description=error) 

86 

87 try: 

88 self.run_in_thread( 

89 AssetDetailPageService.get_asset_transactions, 

90 { 

91 'args': [ListTransfersRequestModel(asset_id=asset_id)], 

92 'callback': on_success, 

93 'error_callback': on_error, 

94 }, 

95 ) 

96 except Exception as e: 

97 on_error(CommonException(message=str(e))) 

98 

99 def on_success_cfa(self, tx_id: SendAssetResponseModel) -> None: 

100 """Handle success for sending CFA asset.""" 

101 self.is_loading.emit(False) 

102 self.send_cfa_button_clicked.emit(False) 

103 ToastManager.success(description=INFO_ASSET_SENT.format(tx_id.txid)) 

104 

105 if self.asset_type == AssetType.CFA.value: 

106 self._page_navigation.collectibles_asset_page() 

107 elif self.asset_type == AssetType.NIA.value: 

108 self._page_navigation.fungibles_asset_page() 

109 

110 def on_error(self, error: CommonException) -> None: 

111 """Handle error for sending CFA asset.""" 

112 self.is_loading.emit(False) 

113 self.send_cfa_button_clicked.emit(False) 

114 ToastManager.error(description=error.message) 

115 

116 def on_success_send_rgb_asset(self, success: bool) -> None: 

117 """Callback function after native authentication is successful.""" 

118 if success: 

119 self.send_cfa_button_clicked.emit(True) 

120 self.is_loading.emit(True) 

121 try: 

122 self.run_in_thread( 

123 RgbRepository.send_asset, 

124 { 

125 'args': [ 

126 SendAssetRequestModel( 

127 asset_id=self.asset_id, 

128 assignment=AssignmentModel( 

129 type=AssignmentEnumModel.FUNGIBLE, value=self.amount, 

130 ), 

131 recipient_id=self.blinded_utxo, 

132 transport_endpoints=self.transport_endpoints, 

133 fee_rate=self.fee_rate, 

134 min_confirmations=self.min_confirmation, 

135 ), 

136 ], 

137 'callback': self.on_success_cfa, 

138 'error_callback': self.on_error, 

139 }, 

140 ) 

141 except Exception as e: 

142 self.on_error(CommonException(message=str(e))) 

143 else: 

144 ToastManager.error(description=ERROR_AUTHENTICATION_CANCELLED) 

145 

146 def on_error_native_auth(self, error: Exception) -> None: 

147 """Callback function on error during native authentication.""" 

148 description = error.message if isinstance( 

149 error, CommonException, 

150 ) else ERROR_SOMETHING_WENT_WRONG 

151 ToastManager.error(description=description) 

152 

153 def on_send_click(self, amount: int, blinded_utxo: str, transport_endpoints: list, fee_rate: int, min_confirmation: int) -> None: 

154 """Starts a thread to execute the send_cfa function with the provided arguments.""" 

155 self.amount = amount 

156 self.blinded_utxo = blinded_utxo 

157 self.transport_endpoints = transport_endpoints 

158 self.fee_rate = fee_rate 

159 self.min_confirmation = min_confirmation 

160 self.run_in_thread( 

161 SettingRepository.native_authentication, 

162 { 

163 'args': [NativeAuthType.MAJOR_OPERATION], 

164 'callback': self.on_success_send_rgb_asset, 

165 'error_callback': self.on_error_native_auth, 

166 }, 

167 ) 

168 

169 def on_refresh_click(self) -> None: 

170 """Executes the refresh operation in a separate thread.""" 

171 cache = Cache.get_cache_session() 

172 if cache is not None: 

173 cache.invalidate_cache() 

174 self.send_cfa_button_clicked.emit(True) 

175 self.is_loading.emit(True) 

176 

177 def on_success_refresh() -> None: 

178 """Handle success for refreshing transactions.""" 

179 self.is_loading.emit(False) 

180 self.refresh.emit(True) 

181 ToastManager.success(description=INFO_REFRESH_SUCCESSFULLY) 

182 self.get_cfa_asset_detail( 

183 self.asset_id, self.asset_name, None, self.asset_type, 

184 ) 

185 

186 def on_error(error: CommonException) -> None: 

187 """Handle error for refreshing transactions.""" 

188 self.refresh.emit(False) 

189 self.is_loading.emit(False) 

190 ToastManager.error( 

191 description=f'{ERROR_SOMETHING_WENT_WRONG}: {error}', 

192 ) 

193 

194 try: 

195 self.run_in_thread( 

196 RgbRepository.refresh_transfer, 

197 { 

198 'args': [], 

199 'callback': on_success_refresh, 

200 'error_callback': on_error, 

201 }, 

202 ) 

203 except Exception as e: 

204 on_error(CommonException(message=str(e))) 

205 

206 def on_fail_transfer(self, batch_transfer_idx: int) -> None: 

207 """Executes the fail transfer operation in a separate thread.""" 

208 self.is_loading.emit(True) 

209 

210 def on_success_fail_transfer(response: FailTransferResponseModel) -> None: 

211 """Handle success for failing a transfer.""" 

212 if response.transfers_changed: 

213 self.get_cfa_asset_detail( 

214 self.asset_id, self.asset_name, None, self.asset_type, 

215 ) 

216 ToastManager.success( 

217 description=INFO_FAIL_TRANSFER_SUCCESSFULLY, 

218 ) 

219 else: 

220 self.is_loading.emit(False) 

221 ToastManager.error(description=ERROR_FAIL_TRANSFER) 

222 

223 def on_error(error: CommonException) -> None: 

224 """Handle error for failing a transfer.""" 

225 self.is_loading.emit(False) 

226 ToastManager.error( 

227 description=f'{ERROR_SOMETHING_WENT_WRONG}: {error.message}', 

228 ) 

229 

230 try: 

231 self.run_in_thread( 

232 RgbRepository.fail_transfer, 

233 { 

234 'args': [FailTransferRequestModel(batch_transfer_idx=batch_transfer_idx)], 

235 'callback': on_success_fail_transfer, 

236 'error_callback': on_error, 

237 }, 

238 ) 

239 except Exception as e: 

240 on_error(CommonException(message=str(e)))
