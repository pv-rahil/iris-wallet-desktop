36class ToasterManager: 

37 """Manages the positioning and lifecycle of toasters.""" 

38 active_toasters: list[ToasterUi] = [] 

39 main_window = None # Keep reference to the main window

41 @classmethod 

42 def set_main_window(cls, main_window): 

43 """Set the main window for the toasters.""" 

44 cls.main_window = main_window

46 @classmethod 

47 def add_toaster(cls, toaster): 

48 """Add a new toaster to the manager and reposition all.""" 

49 if cls.main_window: 

50 cls.active_toasters.append(toaster) 

51 cls.reposition_toasters() 

52 

53 @classmethod 

54 def remove_toaster(cls, toaster): 

55 """Remove a toaster and reposition remaining ones.""" 

56 if toaster in cls.active_toasters: 

57 cls.active_toasters.remove(toaster) 

58 cls.reposition_toasters() 

59 

60 @classmethod 

61 def reposition_toasters(cls): 

62 """Reposition all toasters based on their order.""" 

63 for index, toaster in enumerate(cls.active_toasters): 

64 parent = toaster.parent() 

65 if parent: 

66 # Use geometry() to get the position and size of the parent widget 

67 x = ToasterManager.main_window.size().width() - toaster.width() - 20 

68 y = ToasterManager.main_window.size().height() - (index + 1) * \ 

69 (toaster.height() + int(toaster.height() * 0.1)) - 15 

70 

71 toaster.move(QPoint(x, y))
